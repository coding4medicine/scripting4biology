# For loops


## python

Multiplication table  -

~~~~~~~~
#!/usr/bin/env python

#
# The keyword 'for' is used to loop over the same code many times.
# It uses two other keywords 'in' and 'range'. 
#
for i in range(1,10):
    print i,7*i

~~~~~~~~


## GO

Multiplication table  -

~~~~~~~~
package main
import "fmt"

func main() {
    var i int
    i=1
   for i<=10 {
        fmt.Println(7*i)
        i=i+1
   }
}
